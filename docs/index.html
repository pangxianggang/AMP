2. 快速开始

创建第一个 Agent

import amp
import asyncio

async def main():
    # 创建 Agent
    agent = amp.Agent(
        name="小助手",
        role="assistant",
        language="zh"
    )
    
    # 存储记忆
    await agent.memory.remember(
        "用户喜欢中文回复",
        type=amp.MemoryType.SEMANTIC,
        importance=10
    )
    
    # 回忆记忆
    memories = await agent.memory.recall("用户偏好")
    for mem in memories:
        print(f"记得：{mem.content}")

asyncio.run(main())

运行结果

记得：用户喜欢中文回复

4. Agent Mesh

创建多 Agent 团队

import amp

async def team_work():
    # 创建团队
    researcher = amp.Agent(name="探索者", role="researcher")
    writer = amp.Agent(name="记录员", role="writer")
    
    # Mesh 网络
    mesh = amp.AgentMesh()
    await mesh.register(researcher)
    await mesh.register(writer)
    
    # 团队任务
    result = await mesh.team_task(
        task={"description": "研究 AI 框架"},
        agents=[researcher, writer]
    )
    
    # 知识传承
    await writer.learn_from(researcher, topic="AI")

5. 集成

CLI-Anything 集成

import amp
import subprocess

async def use_gimp():
    designer = amp.Agent(name="设计师", role="designer")
    
    # 调用 GIMP CLI
    subprocess.run([
        "cli-anything-gimp",
        "project", "new",
        "--width", "1920",
        "--height", "1080"
    ])
    
    # 记录经验
    await designer.memory.remember(
        "1920x1080 是海报最佳尺寸",
        type=amp.MemoryType.PROCEDURAL,
        importance=9
    )

OpenClaw 集成

import amp

async def use_skill():
    agent = amp.Agent(
        name="助手",
        role="assistant",
        enable_openclaw=True
    )
    
    # 使用 serper 搜索技能
    result = await agent.use_skill(
        "serper",
        query="Python 异步编程"
    )

6. API 参考

Agent 类

class Agent:
    def __init__(
        self,
        name: str,
        role: str,
        language: str = "zh",
        storage_dir: Optional[str] = None,
        enable_fm: bool = False,
        enable_memory_bank: bool = False,
        enable_openclaw: bool = False
    )
    
    async def initialize() -> None
    async def act(task: Dict[str, Any]) -> Dict[str, Any]
    async def use_skill(skill_name: str, **kwargs) -> Dict[str, Any]
    async def learn_from(other_agent, topic: str = None) -> int
    async def sleep() -> Dict[str, int]
    status() -> Dict[str, Any]

Memory 类

class AgentMemory:
    async def remember(
        content: str,
        type: MemoryType,
        importance: int = 5,
        tags: List[str] = None
    ) -> Memory
    
    async def recall(
        query: str,
        type: MemoryType = None,
        limit: int = 10
    ) -> List[Memory]
    
    async def consolidate() -> Dict[str, int]
    count() -> int
    stats() -> Dict[str, Any]
