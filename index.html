🚀 快速开始

1. 安装

# 进入项目目录
cd AMP

# 安装依赖
pip install -e ./sdk/python

2. 创建你的第一个 Agent

import asyncio
from amp import Agent, MemoryType

async def main():
    # 创建 Agent
    agent = Agent(name="Ali", role="project_manager", language="zh")

    # 存储记忆
    await agent.memory.remember(
        "用户喜欢中文回复",
        type=MemoryType.SEMANTIC,
        importance=8
    )

    await agent.memory.recall("用户偏好")

asyncio.run(main())
